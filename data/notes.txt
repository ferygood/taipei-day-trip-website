建立 database
create database taipei_travel character set utf8;

使用 database
use taipei_travel;

設計 database
main_table:
id (primary key)
name (景點名稱)
address (地址)
create table main_table (id bigint auto_increment, name varchar(255) not
null, address varchar(255) not null, primary key (id));

image_table:
id (foreign key)
image (景點圖片)
create table image_table (id bigint, img varchar(255) not null, foreign key (id) references main_table(id));
