-登入
>
mysql -u root -p yaomysql86

-建立 database
>
create database website character set utf8;

-使用 database
>
use taipei_travel;

-設計 database
main: id (primary key), name (景點名稱), category, 
description, address (地址), transport, mrt, latitude, longitude, images
>
create table main (id int not null auto_increment, name varchar(255), 
category varchar(255), description longtext, address varchar(255) not null, 
transport longtext, mrt varchar(255), latitude float, longitude float, 
images longtext, primary key (id));

-reset auto_increment
ALTER TABLE main AUTO_INCREMENT=1;

-存取 sql 檔案
mysqldump --databases website -u root -p > taipei_travel_database.sql


-如果想要改資料型態
>
alter table main_table change description description longtext;

-每筆取 12 筆資料
>
select id, name from main where name like "%山%" limit 12 offset 0;
