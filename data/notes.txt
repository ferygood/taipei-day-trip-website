-登入
>
mysql -u root -p yaomysql86



-建立 database
>
create database taipei_travel character set utf8;

-使用 database
>
use taipei_travel;

-設計 database
main_table: id (primary key), name (景點名稱), category, 
description, address (地址), transport, mrt, latitude, longitude
>
create table main_table (id bigint auto_increment, name varchar(255) not
null, category varchar(255), description longtext, 
address varchar(255) not null, transport longtext, 
mrt varchar(255), latitude float, longitude float,
primary key (id));

-image_table: id (foreign key), image (景點圖片)
>
create table image_table (id bigint, img varchar(255) not null, 
foreign key (id) references main_table(id));

-存取 sql 檔案
mysqldump --databases taipei_travel -u root -p > taipei_travel_database.sql


-如果想要改資料型態
>
alter table main_table change description description longtext;
